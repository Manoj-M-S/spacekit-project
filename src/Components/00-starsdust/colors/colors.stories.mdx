import { Meta, ColorPalette, ColorItem, Subtitle } from '@storybook/addon-docs';

const secondaryColors = colorTokens.theme1.colors1;

# Primary Colours

<Subtitle>
  These are the main colours that make up the majority of the colours used in
  the design system.
</Subtitle>

<ColorPalette>
  {Object.keys(colors).map(key => {

  })}

</ColorPalette>

<ColorPalette>
